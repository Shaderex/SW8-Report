\newcommand{\quotewithref}[2] {
    \begin{quote}
        ``#1''
    \end{quote}
    \vspace{-0.8em}
    \hspace{25pt} \emph{\textemdash \textcite{#2}}
    \\
}

\newcommand{\quotewithauthorandref}[4] {
    \begin{quote}
        ``#1'' \ifthenelse{\equal{#4}{true}}{(\emph{translated})}{}
    \end{quote}
    \vspace{-0.8em}
    \hspace{25pt} \emph{\textemdash ``#2''} \parencite{#3}
    \\
}
